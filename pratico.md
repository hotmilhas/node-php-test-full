        fetch(BASE_URL + '/api/transacoes')
            .then(result => {
                result.json().then(transactions => {
                    var _transactions = []

                    for (var i in transactions) {
                        if (transactions[i].realizada)  {
                            _transactions.push({
                                id: transactions[i].id,
                                value: transactions[i].valor,
                                type: transactions[i].valor < 0 ? 'transference' : 'deposit',
                            })
                        }
                    }

                    resolve(_transactions)
                })
            })
            .catch(e => reject(e))
    })
}

    $sql = 'SELECT username, email FROM users WHERE username = :username AND password = :password';
    $statement = $pdo->prepare($sql);
    $statement->execute();
    $result = $statement->fetch(PDO::FETCH_ASSOC);
